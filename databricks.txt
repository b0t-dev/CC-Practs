--Check the Average Price
SELECT
  State,
  AVG(CAST(`2015 median sales price` AS DOUBLE)) AS Avg_Price
FROM read_files(
  '/databricks-datasets/samples/population-vs-price/data_geo.csv',
  format => 'csv',
  header => true,
  inferSchema => true
)
GROUP BY State
ORDER BY Avg_Price DESC;


--Check the Population Estimate and Plot a Bar Chart
SELECT
  State,
  SUM(CAST(`2014 population estimate` AS DOUBLE)) AS Total_Population_2014
FROM read_files(
  '/databricks-datasets/samples/population-vs-price/data_geo.csv',
  format => 'csv',
  header => true,
  inferSchema => true
)
GROUP BY State
ORDER BY Total_Population_2014 DESC;
